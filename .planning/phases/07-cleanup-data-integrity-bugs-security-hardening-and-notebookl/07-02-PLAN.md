```ts
/**
 * Minimal session shape required by `assertCustomerAccess`. Pulled from `requireSession`.
 */
export interface AuthorizedSession {
  user: { id: string; role: string };
}

/**
 * Minimal customer shape required by `assertCustomerAccess`. Use a structural type
 * so callers can pass either a full Prisma Customer or a `select`-narrowed projection.
 */
export interface CustomerOwnershipFields {
  assignedRepId: string | null;
}

/**
 * Rep-ownership authorization for customer-scoped resources.
 *
 * Rule (from Phase 7 D-04 / D-05):
 *   - admin or manager → always allowed
 *   - rep → allowed only if customer.assignedRepId === session.user.id
 *
 * Returns `true` if access is allowed, `false` otherwise.
 *
 * Callers should respond with `NextResponse.json({ error: "Forbidden" }, { status: 403 })`
 * when this returns false. We return a boolean (not throw) so route handlers can keep
 * their try/catch blocks focused on real errors rather than authorization control flow.
 */
export function assertCustomerAccess(
  session: AuthorizedSession,
  customer: CustomerOwnershipFields,
): boolean {
  const role = session.user.role;
  if (role === "admin" || role === "manager") return true;
  if (customer.assignedRepId && customer.assignedRepId === session.user.id) return true;
  return false;
}
```

Do NOT add any other exports. Do NOT touch `authOptions`. Do NOT touch the `requireSession` or `auth` functions.

**Background — why this is a checkpoint task:**

The infographic and slide-deck pipelines upload generated files to a Supabase Storage bucket and then call `supabase.storage.from(bucket).getPublicUrl(path)`. If the bucket is **public**, the resulting URL is permanently world-readable, which is a data leak (any rep's briefing PDF, including customer-facing leave-behinds, is accessible to anyone with the URL). If the bucket is **private**, `getPublicUrl` returns a URL that 401s without auth — and the current code is fine.

There is no `list_buckets` capability in the available Supabase MCP server (`mcp__claude_ai_Supabase__*` only exposes table operations). The only way to verify the bucket ACL is to ask the user to check the Supabase dashboard or run the Supabase CLI locally. So this task PAUSES execution and asks the user to do the verification.
